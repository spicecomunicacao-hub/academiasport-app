  <div class="bg-white p-8 rounded-2xl shadow-md w-96">
    <h1 class="text-2xl font-bold text-center mb-4 text-blue-600">Academia SportFitness</h1>
    <p class="text-center text-gray-600 mb-6">Faça login ou cadastre-se</p>

    <!-- Formulário -->
    <form id="authForm" class="space-y-4">
      <input id="name" type="text" placeholder="Seu nome" class="w-full p-2 border rounded" />
      <input id="email" type="email" placeholder="Seu e-mail" class="w-full p-2 border rounded" required />
      <input id="password" type="password" placeholder="Sua senha" class="w-full p-2 border rounded" required />

      <button type="submit" class="w-full bg-blue-600 text-white p-2 rounded hover:bg-blue-700 transition">
        Entrar / Cadastrar
      </button>
    </form>

    <p id="message" class="text-center mt-4 text-sm"></p>
  </div>

  <script>
    const form = document.getElementById('authForm');
    const message = document.getElementById('message');

    form.addEventListener('submit', (e) => {
      e.preventDefault();

      const name = document.getElementById('name').value;
      const email = document.getElementById('email').value;
      const password = document.getElementById('password').value;

      // Simulação de cadastro/login
      const users = JSON.parse(localStorage.getItem('users') || '{}');

      if (users[email]) {
        if (users[email] === password) {
          message.innerText = `Bem-vindo de volta, ${name || 'Aluno'}!`;
          message.className = "text-center mt-4 text-green-600";
        } else {
          message.innerText = "Senha incorreta!";
          message.className = "text-center mt-4 text-red-600";
        }
      } else {
        users[email] = password;
        localStorage.setItem('users', JSON.stringify(users));
        message.innerText = `Conta criada com sucesso! Bem-vindo, ${name || 'Aluno'}!`;
        message.className = "text-center mt-4 text-green-600";
      }

      form.reset();
    });
  </script>
